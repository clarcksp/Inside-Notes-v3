server {
    listen 80;
    server_name localhost;
    charset utf-8;

    # Localização raiz para servir os arquivos do frontend
    location / {
        root   /usr/share/nginx/html;
        index  index.html;
        # Esta linha é crucial para que Single Page Applications (SPAs) funcionem corretamente.
        # Se uma rota como /visita/123 for acessada diretamente, o Nginx servirá o index.html,
        # e o React Router cuidará do roteamento no lado do cliente.
        try_files $uri $uri/ /index.html;
    }

    # Proxy reverso para as chamadas de API para o serviço de backend.
    # Todas as requisições que começam com /api/ serão encaminhadas.
    location /api/ {
        # 'backend' é o nome do serviço definido no docker-compose.yml, rodando na porta 4000.
        proxy_pass http://backend:4000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Redirecionamento de logs de erro e acesso para o stdout/stderr do contêiner
    # para que possam ser visualizados com 'docker logs'.
    error_log /dev/stderr warn;
    access_log /dev/stdout;
}
